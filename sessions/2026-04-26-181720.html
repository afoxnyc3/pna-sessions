DDL (canonical)


-- Migration: 00NN_tasks_and_events.sql

CREATE TABLE IF NOT EXISTS tasks (
  id              TEXT PRIMARY KEY,
  title           TEXT NOT NULL,
  body            TEXT NOT NULL DEFAULT '',
  status          TEXT NOT NULL DEFAULT 'captured'
                    CHECK (status IN ('captured','todo','in_progress','review','blocked','done')),
  created_by      TEXT NOT NULL,
  assigned_agent  TEXT,
  priority        INTEGER NOT NULL DEFAULT 5,
  created_at      INTEGER NOT NULL,
  updated_at      INTEGER NOT NULL,
  closed_at       INTEGER
);

CREATE INDEX IF NOT EXISTS idx_tasks_status ON tasks(status, priority DESC, created_at DESC);
CREATE INDEX IF NOT EXISTS idx_tasks_agent ON tasks(assigned_agent, status);

CREATE TABLE IF NOT EXISTS task_events (
  id          INTEGER PRIMARY KEY AUTOINCREMENT,
  task_id     TEXT NOT NULL,
  event_type  TEXT NOT NULL
                CHECK (event_type IN
                  ('created','status_change','claim','comment','artifact','mission_linked','mission_completed','reopened')),
  actor       TEXT NOT NULL,
  payload     TEXT NOT NULL DEFAULT '{}',
  created_at  INTEGER NOT NULL,
  FOREIGN KEY (task_id) REFERENCES tasks(id) ON DELETE CASCADE
);

CREATE INDEX IF NOT EXISTS idx_task_events_task ON task_events(task_id, created_at DESC);
CREATE INDEX IF NOT EXISTS idx_task_events_type ON task_events(event_type, created_at DESC);

ALTER TABLE mission_tasks ADD COLUMN task_id TEXT;
CREATE INDEX IF NOT EXISTS idx_mission_task ON mission_tasks(task_id, status);

`list [--status backlog\ kanban\ ] [--agent X] [--json]` Read tasks, optionally filtered
show <id> Print task + event timeline + linked missions
move <id> <status> [--reason R] Update status, write status_change event
claim <id> --agent X Set assigned_agent, status→in_progress, write claim event
comment <id> "<text>" Append comment event

assign <id> --agent X Set assigned_agent without status change
reopen <id> [--reason R] If status=done, set status→in_progress, write reopened event

Mission Completion Hook

Pseudocode patch in scheduler.ts where missions transition to completed:


function onMissionCompleted(missionId: string) {
  const m = db.prepare('SELECT task_id FROM mission_tasks WHERE id = ?').get(missionId);
  if (!m?.task_id) return;
  const t = db.prepare('SELECT status FROM tasks WHERE id = ?').get(m.task_id);
  if (t?.status === 'in_progress') {
    db.prepare('UPDATE tasks SET status = ?, updated_at = ? WHERE id = ?')
      .run('review', now(), m.task_id);
    insertTaskEvent(m.task_id, 'mission_completed', 'system', { mission_id: missionId });
  } else {
    insertTaskEvent(m.task_id, 'mission_completed', 'system',
      { mission_id: missionId, no_move: true, current_status: t?.status });
  }
}

// closed_at is set whenever a task transitions to 'done'.
// The status_change handler is the single owner:
function moveStatus(taskId: string, next: Status, actor: string, reason?: string) {
  const now = Date.now();
  const closedAt = next === 'done' ? now : null;
  db.prepare(`
    UPDATE tasks
    SET status = ?, updated_at = ?, closed_at = COALESCE(?, closed_at)
    WHERE id = ?
  `).run(next, now, closedAt, taskId);
  // If reopening, clear closed_at:
  if (next !== 'done') {
    db.prepare('UPDATE tasks SET closed_at = NULL WHERE id = ? AND status != ?')
      .run(taskId, 'done');
  }
  insertTaskEvent(taskId, 'status_change', actor, { to: next, reason });
}

Agent CLAUDE.md amendment block (canonical, copy verbatim)


## Task Discipline

Every meaningful unit of work has a `tasks` row. Before starting, claim it.

1. **Find or create**: if Clarke or Alex did not give you a task id, create one yourself with
   `node /Users/alex/dev/claudeclaw/dist/task-cli.js add --title "<short>" --created-by $CLAUDECLAW_AGENT_ID "<one-line description>"`.
2. **Claim**: `task-cli claim <id> --agent $CLAUDECLAW_AGENT_ID`.
3. **Narrate**: at every meaningful checkpoint, run `task-cli comment <id> "<what just happened>"`. Aim for 2 to 5 comments per non-trivial task.
4. **Link artifacts**: every design doc, ADR, diagram, retro, or pna-sessions URL is linked back with `task-cli link <id> --artifact <url-or-path>`. The publish script accepts `--task-id` and does this for you.
5. **Close out**: when work is shippable, move to `review` (or let mission completion auto-move it). Clarke approves to `done`.

If you skip the ritual, the work is invisible to the rest of the team and to Mission Control. That is the point.
